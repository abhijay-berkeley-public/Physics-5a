\begin{problem}{Law of Cosines Problem} \

	Prove: \begin{equation}
  a \cdot b = \|a\|\|b\|cos\theta
\end{equation}

Using the Law of Cosines...

	\begin{align*}
		|a|^2 \cdot \abs{b}^2 - 2|a||b|cos\theta \\ &= \abs{\ra{a}-\ra{b}} \\
		\abs{a} \cdot \abs{b} &= \abs{a}\abs{b}cos\theta
	\end{align*}

\end{problem}
